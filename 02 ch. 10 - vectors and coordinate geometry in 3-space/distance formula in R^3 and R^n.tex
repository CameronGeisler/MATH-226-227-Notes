\section*{Distance Formula}

Recall that in the plane, the distance from points $P_1 = (x_1, y_1)$ and $P_2 = (x_2, y_2)$ is given by,
\begin{equation*}
    |P_1P_2| = \sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2}
\end{equation*}

It turns out that this formula naturally extends to point in 3-space. In $\mathbb{R}^3$, let $P_1 = (x_1, y_1, z_1), P_2 = (x_2, y_2, z_2)$. Then, consider the distance from $P_1$ to $P_2$. Let $Q_1 = (x_1, y_2, z_1)$, and $Q_2 = (x_2, y_2, z_1)$. Using the distance formula in $1$ dimension,
\begin{align*}
    |P_1Q_1| & = |y_2 - y_1| \\
    |Q_1Q_2| & = |x_2 - x_1| \\
    |P_2Q_2| & = |z_2 - z_1|
\end{align*}
Then, using the distance formula in 2 dimensions, the distance from $P_1$ to $Q_2$ is,
\begin{align*}
    |P_1Q_2| & = \sqrt{|P_1Q_1|^2 + |Q_1Q_2|^2} \\
    & = \sqrt{|y_2 - y_1|^2 + |x_2 - x_1|^2} \\
    |P_1Q_2| & = \sqrt{(y_2 - y_1)^2 + (x_2 - x_1)^2}
\end{align*}
Then, again using the distance formula in $2$ dimensions, the distance from $P_1$ to $P_2$ is,
\begin{align*}
    |P_1P_2| & = \sqrt{|P_1Q_2|^2 + |Q_2P_2|^2} \\
    & = \sqrt{\left(\sqrt{(y_2 - y_1)^2 + (x_2 - x_1)^2}\right)^2 + |z_2 - z_1|^2} \\
    |P_1P_2| & = \sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2 + (z_2 - z_1)^2}
\end{align*}

Thus,

\begin{theorem}
Let $P_1 = (x_1, y_1, z_1), P_2 = (x_2, y_2, z_2)$ be points in $\mathbb{R}^3$. Then, the distance between $P_1$ and $P_2$ is given by,
\begin{equation*}
    |P_1P_2| = \sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2 + (z_2 - z_1)^2}
\end{equation*}
\end{theorem}

This formula can be extended to $\mathbb{R}^n$. Let $P = (x_1, x_2, \dots, x_n)$, $Q = (y_1, y_2, \dots, y_n)$. Then the distance from P to Q is
\begin{equation*}
    |PQ| = \sqrt{(y_1 - x_1)^2 + (y_2 - x_2)^2 + \dots + (y_n - x_n)^2}
\end{equation*}

\begin{example}
Determine the distance between the points:
\begin{enumerate}[(a)]
    \item $(0,0,0)$ and $(2,-1,-2)$
    \begin{equation*}
        d = \sqrt{2^2 + (-1)^2 + (-2)^2} = \sqrt{4 + 1 + 4} = \sqrt{9} = 3
    \end{equation*}
    \item $(-1, -1, -1)$ and $(1, 1, 1)$
    \begin{equation*}
        d = \sqrt{(1 + 1)^2 + (1 + 1)^2 + (1 + 1)^2} = \sqrt{4 + 4 + 4} = \sqrt{12} = 2 \sqrt{3}
    \end{equation*}
    \item $(1, 1, 0)$ and $(0, 2, -2)$
    \begin{equation*}
        d = \sqrt{(-1)^2 + 1^2 + (-2)^2} = \sqrt{1 + 1 + 4} = \sqrt{6}
    \end{equation*}
    \item $(3, 8, -1)$ and $(-2, 3, -6)$
    \begin{equation*}
        d = \sqrt{(-2 - 3)^2 + (3 - 8)^2 + (-6 + 1)^2} = \sqrt{25 + 25 + 25} = \sqrt{75} = 5 \sqrt{5}
    \end{equation*}
\end{enumerate}
\end{example}
